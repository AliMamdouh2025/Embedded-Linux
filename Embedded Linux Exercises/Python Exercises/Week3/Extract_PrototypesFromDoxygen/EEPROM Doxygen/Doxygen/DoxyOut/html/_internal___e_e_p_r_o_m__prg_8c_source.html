304void EEPROM_write(uint16 u16Address, uint8 ucData)
305{
306 /* Disable Global Interrupts to protect EEPROM*/
307 CLEAR_BIT(SREG,7);
308
309 /* Wait for completion of previous write */
310 while(EECR & (1<<EEWE));
311
312 /* Wait for SPMEN to be cleared(important When using bootloader) */
313#if(BOOTLOADER_STATE == BOOTLOADER_USED)
314 while(SPMCR & (1<<SPMEN));
315#endif
316
317#if(EEPROM_UPPDATE_OPTIMIZER == ENABLE)
318 /* Set up address register */
319 EEARH = u16Address>>8;
320 EEARL = u16Address;
321
322 /* Start eeprom read by writing EERE */
323 SET_BIT(EECR,EERE);
324
325 /*check if value is repeated or not */
326 if(EEDR == ucData)
327 {
328 return;
329 }
330 else
331 {
332 /*Continue writing normally*/
333 }
334
335
336
337#endif

343 EEARL = u16Address;
344 EEDR = ucData;
